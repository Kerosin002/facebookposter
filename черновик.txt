            pyautogui.keyUp('altleft')
        else:
            if postContent[j]=='ó':
                pyautogui.keyDown('altleft')
                pyautogui.press('o')
                pyautogui.keyUp('altleft')
            else:
                if postContent[j]=='ą':
                    pyautogui.keyDown('altleft')
                    pyautogui.press('a')
                    pyautogui.keyUp('altleft')
                else:
                    if postContent[j]=='ę':
                        pyautogui.keyDown('altleft')
                        pyautogui.typewrite('e')
                        pyautogui.keyUp('altleft')
                    else:
                        if postContent[j]=='ś':
                            pyautogui.keyDown('altleft')
                            pyautogui.typewrite('s')
                            pyautogui.keyUp('altleft')
                        else:

                                pyautogui.typewrite('z')
                                pyautogui.keyUp('altleft')
                            else:

                                    pyautogui.typewrite('x')
                                    pyautogui.keyUp('altleft')
                                else:
                                    if postContent[j]=='ć':
                                        pyautogui.keyDown('altleft')
                                        pyautogui.typewrite('c')
                                        pyautogui.keyUp('altleft')
                                    else:
                                        if postContent[j]=='ń':
                                            pyautogui.keyDown('altleft')
                                            pyautogui.typewrite('n')
                                            pyautogui.keyUp('altleft')
                                        else:
                                            pyautogui.typewrite(postContent[j])
